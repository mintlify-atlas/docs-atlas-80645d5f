## Overview

The `campaigns` table tracks daily campaign creation and execution. Each campaign represents a batch of **14,400 profiles** distributed across **80 VAs** for outreach activities.

<Info>
Campaigns have a **7-day lifecycle** and are automatically cleaned up by the backend cleanup job.
</Info>

## Schema Definition

```sql
CREATE TABLE campaigns (
  campaign_id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  campaign_date DATE NOT NULL,
  total_assigned INTEGER NOT NULL,
  status TEXT CHECK (status IN ('pending', 'success', 'failed')),
  base_id TEXT NOT NULL,  -- Multi-tenant isolation key
  created_at TIMESTAMP DEFAULT NOW()
);
```

## Campaign Workflow

### Complete Campaign Creation Flow

```typescript
/**
 * POST /api/daily-selection
 * Creates campaign and selects 14,400 profiles
 */
async function createDailySelection(baseId: string) {
  // 1. Check available usernames
  const { count } = await supabase
    .from('global_usernames')
    .select('*', { count: 'exact', head: true })
    .eq('base_id', baseId)
    .eq('used', false)
  
  if (count < 14400) {
    throw new Error(`Insufficient usernames. Have ${count}, need 14400`)
  }
  
  // 2. Create campaign
  const { data: campaign, error: campaignError } = await supabase
    .from('campaigns')
    .insert({
      campaign_date: new Date().toISOString().split('T')[0],
      total_assigned: 14400,
      status: 'pending',
      base_id: baseId
    })
    .select()
    .single()
  
  // 3. Select unused usernames
  const { data: selectedUsernames } = await supabase
    .from('global_usernames')
    .select('*')
    .eq('base_id', baseId)
    .eq('used', false)
    .limit(14400)
  
  // 4. Mark as used
  const ids = selectedUsernames.map(u => u.id)
  await supabase
    .from('global_usernames')
    .update({ used: true, used_at: new Date().toISOString() })
    .in('id', ids)
    .eq('base_id', baseId)
  
  return campaign
}
```

### Distribution Workflow

```typescript
/**
 * POST /api/distribute/{campaign_id}
 * Distributes profiles to 80 VA tables
 */
async function distributeToVAs(campaignId: string, baseId: string) {
  // Get campaign's selected usernames (marked as used)
  const { data: usernames } = await supabase
    .from('global_usernames')
    .select('*')
    .eq('base_id', baseId)
    .eq('used', true)
    .is('used_at', 'not.null')
    .order('used_at', { ascending: false })
    .limit(14400)
  
  // Distribute: 180 profiles per VA, 80 VAs
  const assignments = []
  const PROFILES_PER_VA = 180
  const NUM_VAS = 80
  
  for (let vaNum = 1; vaNum <= NUM_VAS; vaNum++) {
    const startIdx = (vaNum - 1) * PROFILES_PER_VA
    const vaProfiles = usernames.slice(startIdx, startIdx + PROFILES_PER_VA)
    
    for (let i = 0; i < vaProfiles.length; i++) {
      assignments.push({
        campaign_id: campaignId,
        id: vaProfiles[i].id,
        username: vaProfiles[i].username,
        full_name: vaProfiles[i].full_name,
        va_table_number: vaNum,
        position: i + 1,
        job_id: vaProfiles[i].job_id,
        base_id: baseId,
        status: 'pending'
      })
    }
  }
  
  // Insert assignments
  await supabase.from('daily_assignments').insert(assignments)
}
```

### Airtable Sync

```typescript
/**
 * POST /api/airtable-sync/{campaign_id}
 * Syncs profiles to Airtable VA tables
 */
async function syncToAirtable(campaignId: string, baseId: string) {
  const { data: assignments } = await supabase
    .from('daily_assignments')
    .select('*')
    .eq('campaign_id', campaignId)
    .eq('base_id', baseId)
    .order('va_table_number', { ascending: true })
  
  // Group by VA table
  const vaGroups = {}
  assignments.forEach(a => {
    if (!vaGroups[a.va_table_number]) {
      vaGroups[a.va_table_number] = []
    }
    vaGroups[a.va_table_number].push(a)
  })
  
  // Sync each VA table to Airtable
  for (const [vaNum, profiles] of Object.entries(vaGroups)) {
    await syncVATableToAirtable(vaNum, profiles, baseId)
  }
  
  // Update campaign status
  await supabase
    .from('campaigns')
    .update({ status: 'success' })
    .eq('campaign_id', campaignId)
    .eq('base_id', baseId)
}
```

## Lifecycle Management

### 7-Day Cleanup

Campaigns older than 7 days are automatically cleaned up:

```typescript
/**
 * POST /api/cleanup
 * Runs daily at 2 AM via cron job
 */
async function cleanupOldCampaigns(baseId: string) {
  const sevenDaysAgo = new Date()
  sevenDaysAgo.setDate(sevenDaysAgo.getDate() - 7)
  const cutoffDate = sevenDaysAgo.toISOString().split('T')[0]
  
  // Find campaigns to cleanup
  const { data: oldCampaigns } = await supabase
    .from('campaigns')
    .select('campaign_id')
    .eq('base_id', baseId)
    .lt('campaign_date', cutoffDate)
  
  for (const campaign of oldCampaigns) {
    // Get assigned usernames
    const { data: assignments } = await supabase
      .from('daily_assignments')
      .select('id')
      .eq('campaign_id', campaign.campaign_id)
      .eq('base_id', baseId)
    
    const profileIds = assignments.map(a => a.id)
    
    // Reset 'used' flag (allow reuse after 7 days)
    await supabase
      .from('global_usernames')
      .update({ used: false, used_at: null })
      .in('id', profileIds)
      .eq('base_id', baseId)
    
    // Delete assignments
    await supabase
      .from('daily_assignments')
      .delete()
      .eq('campaign_id', campaign.campaign_id)
      .eq('base_id', baseId)
    
    // Delete campaign
    await supabase
      .from('campaigns')
      .delete()
      .eq('campaign_id', campaign.campaign_id)
      .eq('base_id', baseId)
  }
  
  return { campaigns_cleaned: oldCampaigns.length }
}
```

## Error Handling

### Insufficient Usernames

```typescript
if (availableCount < 14400) {
  await supabase
    .from('campaigns')
    .update({ status: 'failed' })
    .eq('campaign_id', campaignId)
  
  throw new Error(`Only ${availableCount} usernames available, need 14400`)
}
```

### Distribution Failure

```typescript
try {
  await distributeToVAs(campaignId, baseId)
  await supabase
    .from('campaigns')
    .update({ status: 'success' })
    .eq('campaign_id', campaignId)
} catch (error) {
  await supabase
    .from('campaigns')
    .update({ status: 'failed' })
    .eq('campaign_id', campaignId)
  
  throw error
}
```

## Best Practices

<Check>**Verify username availability before creating campaign** - Check that at least 14,400 unused usernames exist</Check>

<Check>**Update status atomically** - Use transactions when updating campaign status and related records</Check>

<Check>**Handle partial failures gracefully** - Mark campaign as failed if any step fails, allow retry</Check>

<Warning>**Always include base_id in queries** - Critical for multi-tenant data isolation</Warning>

<Warning>**Don't delete campaigns manually** - Let the cleanup job handle lifecycle management</Warning>
