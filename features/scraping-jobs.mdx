## Overview

The Scraping Jobs system enables you to manage multiple independent scraping campaigns simultaneously. Each job represents a complete scraping operation for a specific influencer on a specific platform, with full data isolation powered by the `base_id` multi-tenant architecture.

## Job Architecture

### Database Schema

Scraping jobs are stored in the `scraping_jobs` table with the following structure:

```typescript
interface ScrapingJob {
  job_id: string;              // Unique job identifier (UUID)
  influencer_name: string;      // Name of the influencer being scraped
  platform: Platform;           // 'instagram' | 'threads' | 'tiktok' | 'x'
  airtable_base_id: string;    // Airtable base URL for VA distribution
  base_id: string;             // Multi-tenant isolation key (source of truth)
  num_vas: number | null;      // Number of virtual assistants
  status: JobStatus;           // 'active' | 'paused' | 'archived'
  created_at: string;
  updated_at: string | null;
}
```

## Active Job Selection

### BaseContext Provider

The application uses React Context to manage the active job globally:

```typescript
/**
 * Fetch the active scraping job from Supabase
 * 
 * Priority:
 * 1. Job where status = 'active'
 * 2. Most recent job (fallback)
 */
export async function getActiveJob(): Promise<ScrapingJob | null> {
  // First, try to get an active job
  const { data: activeJobs } = await supabase
    .from('scraping_jobs')
    .select('*')
    .eq('status', 'active')
    .order('created_at', { ascending: false })
    .limit(1);

  if (activeJobs && activeJobs.length > 0) {
    return activeJobs[0];
  }

  // Fallback: get the most recent job
  const { data: recentJobs } = await supabase
    .from('scraping_jobs')
    .select('*')
    .order('created_at', { ascending: false })
    .limit(1);

  return recentJobs && recentJobs.length > 0 ? recentJobs[0] : null;
}
```

### Fetching Statistics by Base ID

```typescript
export async function getJobStatisticsByBaseId(
  baseId: string
): Promise<JobStatistics | null> {
  // Get available usernames count (used = false)
  const { count: availableUsernames } = await supabase
    .from('global_usernames')
    .select('*', { count: 'exact', head: true })
    .eq('base_id', baseId)
    .eq('used', false);

  // Get used usernames count
  const { count: usedUsernames } = await supabase
    .from('global_usernames')
    .select('*', { count: 'exact', head: true })
    .eq('base_id', baseId)
    .eq('used', true);

  // Get total usernames count
  const { count: totalUsernames } = await supabase
    .from('global_usernames')
    .select('*', { count: 'exact', head: true })
    .eq('base_id', baseId);

  return {
    totalUsernames: totalUsernames || 0,
    usedUsernames: usedUsernames || 0,
    availableUsernames: availableUsernames || 0,
    totalAssignments: totalAssignments || 0,
    totalScrapeResults: totalScrapeResults || 0
  };
}
```

```typescript
// Update job status
export async function updateJobStatus(
  jobId: string,
  status: JobStatus
): Promise<boolean> {
  const { error } = await supabase
    .from('scraping_jobs')
    .update({ status })
    .eq('job_id', jobId);
    
  return !error;
}

// Convenience methods
export async function pauseJob(jobId: string): Promise<boolean> {
  return updateJobStatus(jobId, 'paused');
}

export async function activateJob(jobId: string): Promise<boolean> {
  return updateJobStatus(jobId, 'active');
}

export async function archiveJob(jobId: string): Promise<boolean> {
  return updateJobStatus(jobId, 'archived');
}
```

## Best Practices

<AccordionGroup>
  <Accordion title="Always use base_id from context">
    Never hardcode or cache `base_id` values. Always retrieve from `useBase()` hook to ensure correct job isolation:
    
    ```typescript
    // ✅ Correct
    const { baseId } = useBase();
    await apiPost('/api/scrape-followers', baseId, data);
    
    // ❌ Wrong
    const hardcodedBaseId = 'app1234';
    await apiPost('/api/scrape-followers', hardcodedBaseId, data);
    ```
  </Accordion>
  
  <Accordion title="Check for active job before operations">
    Always verify a job is active before performing scraping or assignment operations:
    
    ```typescript
    const { baseId, activeJob } = useBase();
    
    if (!baseId) {
      toast({
        title: "No active job",
        description: "Please select an active scraping job first",
        variant: "destructive"

    }
    ```
